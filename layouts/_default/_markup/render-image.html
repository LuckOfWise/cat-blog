{{- /* Render Markdown images as with WebP fallback when a .webp sibling exists in static/. For images outside the WebP-ready dirs (static/images/, /ig/, /ogp/), or when no .webp file exists, fall back to a plain . */ -}} {{- $src := .Destination -}} {{- $alt := .PlainText -}} {{- $title := .Title -}} {{- /* Compute WebP candidate: only for site-relative paths ending in .jpg */ -}} {{- $webpSrc := "" -}} {{- if and (strings.HasPrefix $src "/") (strings.HasSuffix $src ".jpg") -}} {{- $webpRel := replace $src ".jpg" ".webp" -}} {{- $localPath := printf "static%s" $webpRel -}} {{- if os.FileExists $localPath -}} {{- $webpSrc = $webpRel -}} {{- end -}} {{- end -}} {{- if $webpSrc -}} {{- else -}} {{- end -}}
