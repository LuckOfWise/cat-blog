{{- /* Renders the grid item's thumbnail block for a given page. Falls back from front-matter `thumbnail` → markdown image → video shortcode poster → "No Image" placeholder. Adds a play-icon overlay when the post contains a {{< video >}} shortcode. 動画記事は本文に markdown 画像を持たないため、front matter に `thumbnail: /videos/cat-.jpg` を指定すると一覧にサムネが出る。 */ -}} {{- $content := .RawContent -}} {{- $image := "" -}} {{- $isVideo := false -}} {{- /* 1. front matter thumbnail を最優先（動画記事用） */ -}} {{- with .Params.thumbnail -}} {{- if strings.HasPrefix . "/" -}} {{- $image = printf "%s%s" $.Site.BaseURL (strings.TrimPrefix "/" .) -}} {{- else -}} {{- $image = . -}} {{- end -}} {{- end -}} {{- /* 2. 本文先頭の markdown 画像 */ -}} {{- if not $image -}} {{- $imgMatches := findRE `!\[.*?\]\(([^)]+)\)` $content 1 -}} {{- if $imgMatches -}} {{- $rawImage := replaceRE `!\[.*?\]\(([^)]+)\)` "$1" (index $imgMatches 0) -}} {{- if strings.HasPrefix $rawImage "/" -}} {{- $image = printf "%s%s" .Site.BaseURL (strings.TrimPrefix "/" $rawImage) -}} {{- else -}} {{- $image = $rawImage -}} {{- end -}} {{- end -}} {{- end -}} {{- $videoMatches := findRE `\{\{<\s*video[^>]*>\}\}` $content 1 -}} {{- if $videoMatches -}} {{- $isVideo = true -}} {{- if not $image -}} {{- $posterMatches := findRE `poster="([^"]+)"` (index $videoMatches 0) 1 -}} {{- if $posterMatches -}} {{- $rawPoster := replaceRE `poster="([^"]+)"` "$1" (index $posterMatches 0) -}} {{- if strings.HasPrefix $rawPoster "/" -}} {{- $image = printf "%s%s" .Site.BaseURL (strings.TrimPrefix "/" $rawPoster) -}} {{- else -}} {{- $image = $rawPoster -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{ if $image }}
